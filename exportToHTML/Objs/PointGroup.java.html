package Objs;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

public class Objs.NormalDistributions.PointGroup implements Serializable {

    public int times;
    public float numToEval;

    public Objs.NormalDistributions.PointGroup(int times, float numToEval){
        this.times = times;
        this.numToEval = numToEval;
    }

    public static List<Float> toIndividualPoints(Objs.NormalDistributions.PointGroup pg){
        List<Float> points = new ArrayList<>();
        for (Float point : points) {
            points.add(point);
        }
        return points;
    }

}
