package Screens.NormalDistributions;

import Generals.Generals;
import Objs.NormalDistributions.Evaluation;
import Screens.NormalDistributions.NormalDistributions.DeleteNormalDistribution;
import Screens.NormalDistributions.NormalDistributions.Points.AddPointGroup;
import Screens.NormalDistributions.NormalDistributions.Points.AddPointIndividually;
import Screens.NormalDistributions.NormalDistributions.Points.ViewDistributionPoints;

public class ExamineNormDist {

    public static void Show(Objs.NormalDistributions.Evaluation e){

        Generals.clearConsole();
        System.out.println("STATISTIKO - Examine Normal Distribution: ");
        String[] evalData = new String[7];
        try{
            Objs.NormalDistributions.Evaluation.EvaluationData ed = e.getEvalData();
            evalData[0] = "PROMEDIO / MEDIA: " + ed.average;
            evalData[1] = "MEDIANA: " + ed.average;
            evalData[2] = "MODA: " + ed.average;
            evalData[3] = "VARIANZA: " + ed.average;
            evalData[4] = "DESV. EST. DE MUESTRA: " + ed.stdDev;
            evalData[5] = "PEARSONS SKEWMENT: " + ed.pearsonsSkewment;
            evalData[6] = "KURTOSIS: " + ed.kurtosis;
        }catch(Objs.NormalDistributions.Evaluation.MorePointsAreNeeded eMPAN){
            evalData = new String[1];
            evalData[0] = "Necesito mas de 3 puntos para calcular datos...";
        }
        String[] nDistData = {
                ("NOMBRE: " + e.name),
                ("DESCRIPCION: " + e.description),
                ("VARIABLE A MEDIR: " + e.variableName),
                ("TOTAL DE PUNTOS: " + Objs.NormalDistributions.Evaluation.getTotalNumberOfPoints(e)),
                (""),
                ("--- DATOS DE LA EVALUACION ---"),
                ("")
        };
        String[][] toOutput = {nDistData, evalData};
        Generals.printDataSeries(toOutput);
        String[] options = {
                "Atras",
                "Ver",
                "Agregar Puntos Individualmente",
                "Agregar Puntos Agrupados",
                "Borrar Puntos Individualmente",
                "Borrar Puntos Agrupados",
                "Eliminar Evaluaci√≥n"
        };
        Generals.printMenuSelection(options, 0);

        int usrInput = Generals.getUserInputInt(": ");
        switch (usrInput){
            case 0:
                MenuNormDist.Show();
                break;
            case 1:
                ViewDistributionPoints.Show(e);
                break;
            case 2:
                AddPointIndividually.Show(e);
                break;
            case 3:
                AddPointGroup.Show(e);
                break;
            case 4:
            case 5:
                Generals.pressEnterToContinue("NO IMPLEMENTADO AUN");
                ExamineNormDist.Show(e);
                break;
            case 6:
                DeleteNormalDistribution.Show(e);
                break;
            default:
                Generals.printCommandNotRecognized(String.valueOf(usrInput));
                ExamineNormDist.Show(e);
        }
    }

}
